JavaScript Fundamentals Sheet

Quick Reference Guide with Examples

Variables Data Types Operators Control Flow Functions Arrays Objects Classes DOM Async Withdrawal

Variables

var x = 10;   // function-scoped
let y = 20;   // block-scoped
const z = 30; // constant

Data Types

// Primitive
string, number, boolean, null, undefined, symbol, bigint

// Reference
let obj = { name: "Alice", age: 25 };

Operators

5 + 2  // 7
5 == "5"   // true
5 === "5"  // false
true && false // false

Control Structures

if (x > y) {
  console.log("X is bigger");
} else {
  console.log("Y is bigger");
}

for (let i=0; i<3; i++) {
  console.log(i);
}

Functions

function greet(name) {
  return "Hello " + name;
}

const greet2 = (name) => `Hello ${name}`;

Arrays

let nums = [1,2,3];
nums.push(4);
let evens = nums.filter(n => n % 2 === 0);

Objects

let user = {
  name: "Alice",
  greet() { console.log("Hi " + this.name); }
};

Classes

class Person {
  constructor(name) {
    this.name = name;
  }
  sayHi() {
    console.log(`Hi, I am ${this.name}`);
  }
}

DOM Manipulation

let btn = document.getElementById("myBtn");
btn.textContent = "Click Me!";
btn.addEventListener("click", () => alert("Clicked!"));

Promises & Async

let promise = new Promise((resolve) => {
  setTimeout(() => resolve("Done!"), 1000);
});
promise.then(msg => console.log(msg));

async function fetchData() {
  let res = await fetch("https://api.example.com");
  let data = await res.json();
  console.log(data);
}
